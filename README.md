## Install

```bash
npm install --save forecast-library
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'forecast-library'
import 'forecast-library/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
